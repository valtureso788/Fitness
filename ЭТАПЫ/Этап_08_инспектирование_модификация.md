## 4. Обнаруженные нарушения и исправления

### Нарушение 1: Длинные строки в data.js

**До:**
```javascript
const trainings = [{id: 1, title: 'Йога для начинающих', coach: 'Иванова А.С.', datetime: '2026-04-25 10:00', capacity: 15, booked: 8, type: 'group', status: 'scheduled'}];
```

**После:**
```javascript
const trainings = [{
  id: 1,
  title: 'Йога для начинающих',
  coach: 'Иванова А.С.',
  datetime: '2026-04-25 10:00',
  capacity: 15,
  booked: 8,
  type: 'group',
  status: 'scheduled'
}];
```

### Нарушение 2: Отсутствие обработки ошибок

**До:**
```javascript
function bookTraining(id) {
  const training = trainings.find(t => t.id === id);
  training.booked += 1;
}
```

**После:**
```javascript
/**
 * Записывает клиента на тренировку.
 * @param {number} id - ID тренировки
 * @returns {boolean} Успешность записи
 */
function bookTraining(id) {
  const training = trainings.find(t => t.id === id);
  if (!training) {
    showNotification('Тренировка не найдена', 'error');
    return false;
  }
  if (training.booked >= training.capacity) {
    showNotification('Нет свободных мест', 'error');
    return false;
  }
  training.booked += 1;
  showNotification('Вы записаны!', 'success');
  return true;
}
```

### Нарушение 3: Отсутствие валидации форм

**До:**
```javascript
function login() {
  const email = document.getElementById('email').value;
  const password = document.getElementById('password').value;
  authorize(email, password);
}
```

**После:**
```javascript
/**
 * Обрабатывает вход пользователя с валидацией.
 */
function login() {
  const email = document.getElementById('email').value.trim();
  const password = document.getElementById('password').value;

  if (!email || !password) {
    showNotification('Заполните все поля', 'error');
    return;
  }
  if (!isValidEmail(email)) {
    showNotification('Некорректный email', 'error');
    return;
  }
  authorize(email, password);
}
```
